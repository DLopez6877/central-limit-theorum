pop1 = np.random.binomial(10, 0.3, 10000)
pop2 = np.random.binomial(10,0.5, 10000)

sample1 = np.random.choice(pop1, 100, replace=True)
sample2 = np.random.choice(pop2, 100, replace=True)

plt.hist(sample1, alpha=0.5, label='Sample 1', color="chartreuse") 
plt.hist(sample2, alpha=0.5, label='Sample 2', color="midnightblue") 
plt.legend(loc='upper right') 
plt.show()

from scipy.stats import ttest_ind
print(ttest_ind(sample2, sample1, equal_var=False))

Ttest_indResult(statistic=7.3880631267064603, pvalue=4.1270825036512524e-12)

The T-statistic decreases and the pvalue increases when the samples become more similar.
